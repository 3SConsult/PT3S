def plot():

    fig, ax0 = plt.subplots(figsize=Rm.DINA3q)

    ax0.set_yticks(np.linspace(0, 10, 21))
    ax0.yaxis.set_ticklabels([])
    ax0.grid()

    #PH
    ax1 = ax0.twinx()
    fyPH(ax1)
    PH_SL=ax1.plot(dfAGSN['LSum'], dfAGSN['PH_n'], color='red', label='PH SL',ls='dotted')
    PH_RL=ax1.plot(dfAGSNRL['LSum'], dfAGSNRL['PH_n'], color='blue', label='PH RL',ls='dotted')

    #mlc
    ax11 = ax0.twinx()
    fymlc(ax11)
    mlc_SL=ax11.plot(dfAGSN['LSum'], dfAGSN['mlc_n'], color='red', label='mlc SL')
    mlc_RL=ax11.plot(dfAGSNRL['LSum'], dfAGSNRL['mlc_n'], color='blue', label='mlc RL')

    z=ax11.plot(dfAGSN['LSum'], dfAGSN['ZKOR_n'], color='black', label='z',ls='dashed',alpha=.5)

    #barBZG
    ax12 = ax0.twinx()
    fybarBzg(ax12)
    barB_SL=ax12.plot(dfAGSN['LSum'], dfAGSN['H_n'], color='red', label='H SL',ls='dashdot')
    barB_RL=ax12.plot(dfAGSNRL['LSum'], dfAGSNRL['H_n'], color='blue', label='H RL',ls='dashdot')

    #M
    ax2 = ax0.twinx()
    fyM(ax2)
    QM_SL=ax2.step(dfAGSN['LSum'], dfAGSN['QM']*dfAGSN['direction'], color='orange', label='M SL')
    QM_RL=ax2.step(dfAGSNRL['LSum'], dfAGSNRL['QM']*dfAGSNRL['direction'], color='cyan', label='M RL',ls='--')

    #T
    ax3 = ax0.twinx()
    fyT(ax3)
    T_SL=ax3.plot(dfAGSN['LSum'], dfAGSN['T_n'], color='pink', label='T SL')
    T_RL=ax3.plot(dfAGSNRL['LSum'], dfAGSNRL['T_n'], color='lavender', label='T RL')

    ax0.set_title('Longitudinal Section for '+dbFilename)

    # added these three lines
    lns =  PH_SL+ PH_RL + mlc_SL+ mlc_RL + barB_SL+ barB_RL+ QM_SL+ QM_RL + T_SL+ T_RL + z
    labs = [l.get_label() for l in lns]
    ax0.legend(lns, labs)#, loc=0)

    plt.show()
